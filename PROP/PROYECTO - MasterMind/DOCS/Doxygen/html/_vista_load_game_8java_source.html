267 private void loadGames() {
268 listGames = cLoadGame.getSavedGames();
269
270 if (listGames.size() == 0) {
271 buttonContinue.setEnabled(false);
272 JLabel noGames = new JLabel("No games saved");
273 noGames.setBounds(44, 160, 150, 20);

279 valors = new String[listGames.size()];
280
281 for (int i = 0; i < listGames.size(); i++) {
282 Integer id = listGames.get(i).getIdPartida();
283 String rol = listGames.get(i).getRol();
284 Integer dif = listGames.get(i).getDificultat();
285 String dificultat = "";
286
287 if (dif.equals(DIFICULTY_EASY)) dificultat = "Easy";
288 else if (dif.equals(DIFICULTY_MEDIUM)) dificultat = "Medium";
289 else if (dif.equals(DIFICULTY_HARD)) dificultat = "Hard";
290
291 int difAux = 0;
292 if (dif.equals(DIFICULTY_EASY) || dif.equals(DIFICULTY_HARD)) difAux = 4;
293
294 int aux = 0;
295 if (rol.equals("CodeBreaker")) aux = 1;
296
297 valors[i] = " " + id + " ".repeat(43-aux) + rol + " ".repeat(25+difAux) + dificultat;
298 }
